<script>
   function togglePersistence(persistence) {
        
    // filter out none
        if(persistence == 'none') {
            
            hideObj('pool_persistence_timeout');
            hideObj('pool_persistence_timeout_fallback');

            return;
        } 

        var persistType = document.forms[0].elements[(persistence + "_type")].value;
        var persistValue =  document.forms[0].elements[(persistence + "_hidden_timeout")].value;

        if(persistType == '1') { 
            
            showObj('pool_persistence_timeout');
            hideObj('pool_persistence_timeout_fallback');
            findObj('simple_timeout' + "_SELECT").options[0].value = persistValue;
            findObj('simple_timeout' + "_SELECT").options[0].text = persistValue;

        } else {

            hideObj('pool_persistence_timeout');
            showObj('pool_persistence_timeout_fallback');
            findObj('simple_timeout'  + "_FALLBACK").value = persistValue;
        }
    }
</script>
